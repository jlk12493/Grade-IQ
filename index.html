// GradeIQ Cloudflare Worker // Routes: / and /landing -> landing.html // /app and /app/* -> index.html // everything else -> index.html (fallback) export default { async fetch(request, env) { const url = new URL(request.url); const path = url.pathname; // Route to landing page if (path === '/' || path === '/landing' || path === '/landing.html') { const landing = await env.ASSETS.fetch(new Request(new URL('/landing.html', request.url))); return new Response(landing.body, { headers: { 'Content-Type': 'text/html;charset=UTF-8', 'Cache-Control': 'public, max-age=300', } }); } // Route to app if (path === '/app' || path === '/app/' || path.startsWith('/app/')) { const app = await env.ASSETS.fetch(new Request(new URL('/index.html', request.url))); return new Response(app.body, { headers: { 'Content-Type': 'text/html;charset=UTF-8', 'Cache-Control': 'no-cache', } }); } // Fallback - serve index.html for anything else const app = await env.ASSETS.fetch(new Request(new URL('/index.html', request.url))); return new Response(app.body, { headers: { 'Content-Type': 'text/html;charset=UTF-8', 'Cache-Control': 'no-cache', } }); } }
